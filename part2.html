Methods to count parts of speech¶

In [3]:
def count_nouns(doc):
    return len([w for w in list(doc) if w.tag_.startswith('N')])
def count_adjectives(doc):
    return len([w for w in list(doc) if w.tag_.startswith('J')])
def count_verbs(doc):
    return len([w for w in list(doc) if w.tag_.startswith('V')])

from glob import glob
import random
from numpy import *


SIZE = 1000


fnames = glob('cases/*txt')
random.shuffle(fnames) # randomizes the list 
fnames = fnames[:SIZE] # chose 1000 documents

posCount = zeros((SIZE,4)) # array containting parts of speachs: order [nouns,verbs,adjectives,totalWords]
years = zeros(SIZE)

for i,fname in enumerate(fnames):
    docname = fname.split('/')[-1][:-4]
    year,caseid = docname.split('_')
    years[i] = year
    text = open(fname).read()
    doc = nlp(text)
    posCount[i,0] = count_nouns(doc)
    posCount[i,1] = count_verbs(doc)
    posCount[i,2] = count_adjectives(doc)
    posCount[i,3] = len(doc)
    if i % 50 == 0:
        print(i) # progress indicator

Creating plot of POS frequency per year¶

In [17]:
y0 = int(min(years))
yend = int(max(years))
span = yend-y0
posperyear = zeros((span,4))
for year in range(y0,yend):
    index = (years == year)
    posperyear[year-y0,:] = sum(posCount[index,:],axis=0)

import matplotlib.pyplot as plt
x = arange(y0,yend)
plt.plot(x,posperyear[:,0],label='nouns')
plt.plot(x,posperyear[:,1],label='verbs')
plt.plot(x,posperyear[:,2],label='adjectives')
plt.plot(x,posperyear[:,3],label='total words')
plt.legend()
plt.xlabel('Year')
plt.ylabel('Number of Words')
plt.savefig('POS',dpi=1000)
plt.show()
